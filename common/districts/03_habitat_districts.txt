@hab_cost = 500
@hab_maintenance = 2

# housing adds 12 housing + 4 clerk jobs; or 2 maint. drone jobs
district_hab_housing = {
	base_buildtime = 240
	is_capped_by_modifier = no
	
	show_on_uncolonized = {
		is_planet_class = pc_habitat
	}
	
	potential = {
		is_planet_class = pc_habitat
	}
	
	resources = {
		category = planet_districts
		cost = {
			minerals = @hab_cost
		}
		upkeep = {
			energy = @hab_maintenance
		}		
	}
	
	planet_modifier = {
		planet_housing_add = 12
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			job_maintenance_drone_add = 2
		}
	}
	
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { 
				is_fallen_empire = no
				is_regular_empire = yes 
			}
		}
		modifier = {
			job_clerk_add = 4
		}
	}
	
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { 
				is_fallen_empire = no
				is_regular_empire = yes 
			}
		}
		text = job_clerk_effect_desc
	}
	
	triggered_desc = {
		trigger = { 
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_maintenance_drone_effect_desc
	}	
	
	ai_weight = {
		weight = 1
		modifier = {
			factor = 12
			free_housing < 0
		}
	}
}

			}

		modifier = {
			job_technician_add = 4
		}
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_fallen_empire_spiritualist = yes }
		}
		modifier = {
			job_fe_acolyte_generator_add = 4
		}
	}

	triggered_desc = {
		text = job_technician_drone_effect_desc
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { 
				is_gestalt = no
				is_fallen_empire_spiritualist = no
			}
		}
		text = job_technician_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_fallen_empire_spiritualist = yes }
		}
		text = job_fe_acolyte_generator_effect_desc
	}

		tech_power_plant_1
	}

		weight = 1
		modifier =

			factor = 3

				has_monthly_income = {
					resource = energy
					value < 8
				}
			}
		}

		modifier = {
			factor = 2
			free_housing < 0
		}

		modifier = {
			factor = 3
			any_owned_pop = { 
				OR = { 
					is_unemployed = yes 
					has_job = servant 
				}
			}
		}

		# No one to work it anyway
		modifier = {
			factor = 0
			free_housing > 0
			NOT = {
				any_owned_pop = { 
					OR = { 
						is_unemployed = yes 
						has_job = servant 
					}
				}
			}
		}
	}

	ai_resource_production = {
		energy = 1
		trigger = {
			always = yes
		}
	}
