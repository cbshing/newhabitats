# Habitat General
# Does Nothing, but Overwrites Default Habitat Colony Type
col_habitat = {
	icon = 12

	potential = {
		is_planet_class = pc_habitat
	}
	
	weight_modifier = {
		modifier = {
			add = 500
			NOR = { # No planet specialization at all
				has_planet_flag = energy_habitat
				has_planet_flag = foundry_habitat
				has_planet_flag = trade_habitat
				has_planet_flag = cultural_habitat
				has_planet_flag = research_habitat
				has_planet_flag = void_habitat
			}
			num_districts = { type = district_hab_housing value < 4 }
		}
	}
}

# Residential Habitat
# Increases Pop Growth Speed by 20%, Increases Planet Amenities by 5
col_habitat_residential = {
	icon = 12

	potential = {
		is_planet_class = pc_habitat
		exists = owner
		owner = { is_regular_empire = yes }
	}
	
	planet_modifier = {
		pop_growth_speed = 0.2
		planet_amenities_add = 5
	}

	weight_modifier = {
		modifier = {
			add = 10
			NOR = { # No planet specialization
				has_planet_flag = energy_habitat
				has_planet_flag = foundry_habitat
				has_planet_flag = trade_habitat
				has_planet_flag = cultural_habitat
				has_planet_flag = research_habitat
				has_planet_flag = void_habitat
			}	
			num_districts = { type = district_hab_housing value >= 4 }
		}
	}
}
